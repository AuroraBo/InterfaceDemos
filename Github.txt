git init 初始化创建本地仓库 
# 工作区
# 暂缓区
# 历史提交区
# push
git clone 地址 克隆文件到本地
git config --list 本地git配置
git config --global user.name [name] 设置用户名
git config --global user.email [email] 设置邮箱
git add [file] 提交单个文件
git add . 提交仓库内所有文件
git status 查看提交内容
git commit -m "this is first" 提交到历史提交区
git remote 查看git链接
git remote add origin [地址] 创建链接
git push origin master 本地仓库推送到远程仓库
git commit 提交
